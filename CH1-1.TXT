SQL -- STRUCTURE QUERY LANGUAGE 

VERTICAL : COLUMNS

HORIZONTAL : ROWS

COMBINATION : TABLE

SELECT * FROM EMPLOYEES;

BASIC SELECT STATEMENT

 * : META CHARACTER

SQL IS CASE INSENSITIVE LANGUAGE

CASE SENSITIVE -- a,A
UPPERCASE CHARACTER AND LOWERCASE CHARACTER IS DIFFERENT 
    
CASE INSENSITIVE -- 
UPPERCASE CHARACTER AND LOWERCASE CHARACTER IS  NOT DIFFERENT 

SELECT STATEMENT IS USED TO DISPLAY DATA STORED IN TABLE

SELECT STATEMENT HAS 4 PARTS.

SYNTAX:SELECT COLUMN_NAME FROM TABLE_NAME;

SELECT * FROM EMPLOYEES;

SELECT FIRST_NAME FROM EMPLOYEES;

SELECT LAST_NAME FROM EMPLOYEES;

SELECT SALARY FROM EMPLOYEES;
