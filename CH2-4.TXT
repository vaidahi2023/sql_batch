DEPARTMENT_ID -- 10,50

SELECT FIRST_NAME,DEPARTMENT_ID FROM EMPLOYEES WHERE DEPARTMENT_ID=ANY(10,50);

SELECT FIRST_NAME,DEPARTMENT_ID FROM EMPLOYEES WHERE DEPARTMENT_ID=SOME(10,50);

SELECT * FROM EMPLOYEES WHERE MANAGER_ID=SOME(101,102,103) ORDER BY MANAGER_ID;
