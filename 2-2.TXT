SQL> 
SQL> REM SYNTAX OF SELECT : SELECT ... FROM ... WHERE ... ORDER BY ... OFFSET ... FETCH ...;
SQL> 
SQL> REM OFFSET AND FETCH ARE USED FOR TOP n ANALYSIS

DEFAULT OFFSET VALUE IS 0

SELECT * FROM EMPLOYEES FETCH FIRST 5 ROWS ONLY;

SELECT * FROM EMPLOYEES OFFSET 5 ROWS;

1 TO DISPLAY ROWS FROM TOP OF THE TABLE (DISPLAY FIRST 5 ROWS OF TABLE)

SELECT * FROM EMPLOYEES FETCH FIRST 5 ROWS ONLY;

TO DISPLAY ROWS FROM BOTTOM OF THE TABLE (DISPLAY LAST 7 ROWS OR DISPLAY ALL ROWS STARTING FROM 12)

SELECT * FROM EMPLOYEES OFFSET 13 ROWS;

TO DISPLAY ROWS FROM THE MIDDLE OF THE TABLE (DISPLAY ROWS BETWEEN 5 AND 11)

SELECT * FROM EMPLOYEES OFFSET 4 ROWS FETCH FIRST 7 ROWS ONLY;


TO DISPLAY ROWS FROM THE MIDDLE OF THE TABLE (DISPLAY ROWS BETWEEN 12 AND 18)

SELECT * FROM EMPLOYEES OFFSET 11 ROWS FETCH FIRST 7 ROWS ONLY;

DISPLAY LAST 10 ROWS 

SELECT * FROM EMPLOYEES OFFSET 10 ROWS;

DISPLAY LAST 12 ROWS

select * from employees offset 8 ROWS

DISPLAY ROWS BETWEEN 3 AND 11

select * from employees offset 2 row fetch first 9 row only;

DISPLAY ROWS BETWEEN 8 AND 18

select * from employees offset 7 row fetch 11 row only;
